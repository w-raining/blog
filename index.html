git

git的使用

git的命令行

git checkout -b user   创建并切换user分支
git branch user        创建分支
git checkout user      切换分支
------------------------------------------
git last               查看最后一条记录
git log                用于查看历史纪录
------------------------------------------
解决冲突
git status             查看状态 git会告诉我们冲突的文件
git checkout -b aa     创建一个新的分支 并切换

git add aa.html        提交
git commit -m '1.0'
git push   
git branch -d aa
git log                查看分支的合并情况

git br -d aa           删除aa分支  
-------------------------------------------
git pull               抓取代码

多人协作

查看远程库信息，使用git remote -v；

本地新建的分支如果不推送到远程，对其他人就是不可见的；

从本地推送分支，使用git push origin branch-name，如果推送失败，先用git pull抓取远程的新提交；

在本地创建和远程分支对应的分支，使用git checkout -b branch-name origin/branch-name，本地和远程分支的名称最好一致；

建立本地分支和远程分支的关联，使用git branch --set-upstream branch-name origin/branch-name；

从远程抓取分支，使用git pull，如果有冲突，要先处理冲突。

ajax的使用

$.ajaxSetup({
    headers: {
        'X-CSRF-TOKEN': '{{ csrf_token() }}'
    }
});
$(function () {
    $('#name').focus();
    var reg = $("#register").Validform({
        tiptype: function (msg, o, cssctl) {
            if (!o.obj.is("form")) {
                var objtip = o.obj.siblings(".Validform_label");
                cssctl(objtip, o.type);
                objtip.text(msg);
            }
        },
        label: ".Validform_label",
        showAllError: true,
        ajaxPost: true,
        datatype: {
            "a11": /^1[34578]\d{9}$/
        }
    });

    reg.addRule([
        {
            ele: "#mobile",
            datatype: "a11",
            nullmsg: "请输入手机号！",
            errormsg: "请正确输入手机号！"
        },
        {
            ele: "#captcha",
            datatype: "n4-6",
            nullmsg: "请输入验证码！",
            errormsg: "请正确输入验证码！"
        },
        {
            ele: "#password",
            datatype: "*6-12",
            nullmsg: "请输入密码！",
            errormsg: "请输入6-12个字符密码"
        },
        {
            ele: "#rep",
            datatype: "*",
            nullmsg: "请再次输入密码！",
            errormsg: "两次输入密码不正确",
            recheck: "userpassword"
        },
        {
            ele: "#num",
            datatype: "*",
            nullmsg: "请输入验证码！",
            errormsg: "验证码格式不正确"
        }
    ]);

    var regis = $("#register").Validform();


    var phone = $('#mobile').val();

    //点击刷新验证码
    $('.photo>img').click(function () {
        var timestamp = new Date().getTime();
        $(this).attr("src", '/captcha/default?' + timestamp);
    });
    //倒计时60s
    var wait = 60;

    function time(o) {
        if (wait === 0) {
            o.removeAttribute("disabled");
            o.value = "发送验证码";
            wait = 60;
        } else {
            o.setAttribute("disabled", true);
            o.value = "重新发送(" + wait + ")";
            wait--;
            setTimeout(function () {
                time(o);
            }, 1000)
        }
    }

    $("input[name=phone]").bind('input porpertychange', function () {
        var regex = /^1[34578]\d{9}$/;
        if (regex.test($('input[name=phone]').val())) {
            $('#sent_message_code').attr('disabled', false);
        }
    });
    $('#sent_message_code').click(function () {
        var obj = this;
        var phone = $('input[name=phone]').val();
        $.ajax({
            type: 'POST',
            url: '/login/send',
            dataType: 'json',
            data: {phone: phone},
            async: true,
            success: function (data) {
                if (data.code === 0) {
                    time(obj);
                    $('#div-errors').hide();
                } else {
                    $('#error-tip').text(data.msg);
                    $('#div-errors').show();
                    console.log(data);
                }
            }
        });
    });
    $('#submit-btn').click(function () {
        $.ajax({
            type: 'POST',
            url: '/login/adduser',
            dataType: 'json',
            data: $('#register').serializeJson(),
            async: true,
            success: function (data) {
                if (data.code === 0) {  //注册成功 跳转到成功页面
                    location.href = '/login/rsc';
                } else {
                    $('#error-tip').text(data.msg);
                    $('#div-errors').show();
                    console.log(data);
                }
            }
        });
    });
})
